def hourly(x):
    if x in [0,1,2,3]:
        return 1
    elif x in [4, 5,6,7]:
        return 2
    elif x in [8,9,10,11]:
        return 3
    elif x in [12,13,14,15]:
        return 4
    elif x in [16,17,18,19]:
        return 5
    else:
        return 6
